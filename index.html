DROP TABLE IF EXISTS membros CASCADE; -- RECRIAR COM NOMES EM INGLÊS CREATE TABLE membros ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, av TEXT UNIQUE NOT NULL, nome TEXT NOT NULL, cargo TEXT, dept TEXT, cor TEXT DEFAULT '#4F46E5', email TEXT, whatsapp TEXT, online BOOLEAN DEFAULT false, created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE espacos ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, nome TEXT NOT NULL, estacao TEXT, icone TEXT DEFAULT '📁', descricao TEXT, created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE projetos ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, nome TEXT NOT NULL, dept TEXT, descricao TEXT, prior TEXT DEFAULT 'media', status TEXT DEFAULT 'ativo', prog INTEGER DEFAULT 0, fim TEXT, eq TEXT[], espaco_id UUID REFERENCES espacos(id), created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE tarefas ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, titulo TEXT NOT NULL, projeto_id UUID REFERENCES projetos(id), resp TEXT, prior TEXT DEFAULT 'media', status TEXT DEFAULT 'to do', ok BOOLEAN DEFAULT false, prazo TEXT, descricao TEXT, tempo_gasto INTEGER DEFAULT 0, data_inicio TEXT, created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE riscos ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, descricao TEXT, sev TEXT DEFAULT 'Média', resp TEXT, status TEXT DEFAULT 'aberto', projeto_id UUID REFERENCES projetos(id), created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE reunioes ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, tit TEXT NOT NULL, dia TEXT, mes TEXT, hora TEXT, local TEXT, link TEXT, participantes TEXT[], status TEXT DEFAULT 'proxima', ata TEXT DEFAULT '', created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE agenda ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, tit TEXT NOT NULL, tipo TEXT, status TEXT DEFAULT 'pendente', ic TEXT DEFAULT '📋', created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE ferias ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, membro_av TEXT NOT NULL, inicio TEXT NOT NULL, fim TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE atividades ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, tipo TEXT, titulo TEXT, descricao TEXT, autor TEXT, projeto_id UUID, tarefa_id UUID, cor TEXT DEFAULT '#4F46E5', created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE notificacoes ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, destinatario TEXT NOT NULL, titulo TEXT NOT NULL, texto TEXT, tipo TEXT DEFAULT 'info', lida BOOLEAN DEFAULT false, projeto_id UUID, tarefa_id UUID, created_at TIMESTAMPTZ DEFAULT now() );
